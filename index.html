<script>  
const SCRIPT_URL = "PASTE_YOUR_WEB_APP_URL_HERE";  
function showPage(page) {  
document.querySelectorAll(".page").forEach(p =>  
p.classList.remove("active"));  
document.getElementById(page).classList.add("active");  
if (page === "passport") loadPassport();  
}
function apply() {  
const name = document.getElementById("name").value;  
const country = document.getElementById("country").value;  
fetch(SCRIPT_URL, {  
method: "POST",  
body: JSON.stringify({ name, country })  
})  
.then(res => res.json())  
.then(data => {  
localStorage.setItem("name", name);  
localStorage.setItem("id", data.id);  
document.getElementById("result").innerText = "ID: " + data.id;  
});  
}
function loadPassport() {  
const name = localStorage.getItem("name");  
const id = localStorage.getItem("id");  
if (name && id) {  
document.getElementById("passportBox").innerHTML = `  
<h3>  
Passport</h3>  
<p>Name: ${name}</p>  
<p>ID: ${id}</p>  
<p>Nation: Floptropica</p>  
`;  
} else {  
document.getElementById("passportBox").innerText = "No passport found.";  
}
}

setTimeout(() => {

}, 8000);  
};
